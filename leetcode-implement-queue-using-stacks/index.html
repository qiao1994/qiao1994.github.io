type stack struct {
    val []int
}

func (this *stack) push(val int) {
    this.val = append([]int{val}, this.val...)
}

func (this *stack) pop() int {
    result := this.val[0]
    this.val = this.val[1:]
    return result
}

func (this *stack) peek() int {
    return this.val[0]
}

func (this *stack) size() int {
    return len(this.val)
}

func (this *stack) empty() bool {
    if (0 == len(this.val)) {
        return true
    }
    return false
}
